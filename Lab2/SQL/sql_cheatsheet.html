SQL Cheat Sheet

DDL Database
إنشاء / حذف / استخدام
CREATE DATABASE school;
DROP DATABASE   school;
USE             school;
DDL Create Table 
CREATE TABLE students (
  id       INT          PRIMARY KEY AUTO_INCREMENT,
  name     VARCHAR(100) NOT NULL,
  grade    FLOAT        DEFAULT 0,
  dept_id  INT,
  FOREIGN KEY (dept_id) REFERENCES departments(id)
);
DDL Alter Table
إضافة عمود
ALTER TABLE students ADD email VARCHAR(200);
حذف عمود
ALTER TABLE students DROP COLUMN email;
تغيير نوع عمود
ALTER TABLE students MODIFY grade DECIMAL(5,2);
تغيير اسم عمود
ALTER TABLE students
  RENAME COLUMN grade TO score;
تغيير اسم الجدول
RENAME TABLE students TO pupils;
DML Insert
إضافة صف واحد
INSERT INTO students (name, grade, dept_id)
VALUES ('Ali', 89.5, 1);
إضافة عدة صفوف
INSERT INTO students (name, grade, dept_id)
VALUES
  ('Sara', 92, 2),
  ('Omar', 75, 1);
DML Update
تعديل قيمة طالب معين
UPDATE students
SET    grade = 95
WHERE  id = 3;
تعديل عدة أعمدة
UPDATE students
SET    grade = 95, dept_id = 2
WHERE  name = 'Ali';
DML Delete
حذف صف محدد
DELETE FROM students WHERE id = 3;
حذف كل الصفوف (مع الحفاظ على الجدول)
DELETE FROM students;
-- أو أسرع:
TRUNCATE TABLE students;
حذف الجدول كله
DROP TABLE students;
DQL Select 
SELECT *               FROM students;
SELECT name, grade     FROM students;
SELECT DISTINCT dept_id FROM students;

-- alias
SELECT name AS student_name FROM students;

-- شرط
SELECT * FROM students WHERE grade > 80;

-- BETWEEN / IN / LIKE / IS NULL
WHERE grade BETWEEN 70 AND 90
WHERE dept_id IN (1, 2, 3)
WHERE name LIKE 'A%'
WHERE email IS NULL
DQL Order By / Limit 
SELECT * FROM students
ORDER BY grade DESC;     -- أو ASC

-- أول 5 نتائج
SELECT * FROM students
ORDER BY grade DESC
LIMIT 5;

-- تخطي أول 10 وجيب 5
LIMIT 5 OFFSET 10;
DQL Group By / Having 
-- عدد الطلاب في كل قسم
SELECT   dept_id, COUNT(*) AS total
FROM     students
GROUP BY dept_id;

-- متوسط الدرجة لكل قسم + فلتر
SELECT   dept_id, AVG(grade) AS avg_grade
FROM     students
GROUP BY dept_id
HAVING   AVG(grade) > 75; -- HAVING بعد GROUP BY
Functions Aggregate Functions 
COUNT(*)        -- عدد الصفوف
SUM(grade)      -- مجموع
AVG(grade)      -- متوسط
MAX(grade)      -- أكبر قيمة
MIN(grade)      -- أصغر قيمة

-- مثال كامل
SELECT
  COUNT(*)   AS total,
  AVG(grade) AS avg,
  MAX(grade) AS highest
FROM students;
JOIN Joins — كل الأنواع
INNER JOIN — الصفوف المشتركة فقط
SELECT s.name, d.dept_name
FROM   students s
INNER JOIN departments d
  ON s.dept_id = d.id;
LEFT JOIN — كل students + match من departments
SELECT s.name, d.dept_name
FROM   students s
LEFT JOIN departments d
  ON s.dept_id = d.id;
RIGHT JOIN — كل departments + match من students
SELECT s.name, d.dept_name
FROM   students s
RIGHT JOIN departments d
  ON s.dept_id = d.id;
FULL OUTER JOIN — الكل
SELECT s.name, d.dept_name
FROM   students s
FULL OUTER JOIN departments d
  ON s.dept_id = d.id;
CROSS JOIN — كل combination ممكن
SELECT s.name, d.dept_name
FROM   students s
CROSS JOIN departments d;
SELF JOIN — الجدول مع نفسه
SELECT a.name, b.name AS buddy
FROM   students a
JOIN   students b
  ON a.dept_id = b.dept_id
  AND a.id != b.id;
Keys Primary Key / Foreign Key 
-- Primary Key عند الإنشاء
CREATE TABLE departments (
  id   INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100)
);

-- Foreign Key (يشير لـ PK جدول ثاني)
CREATE TABLE students (
  id      INT PRIMARY KEY AUTO_INCREMENT,
  dept_id INT,
  FOREIGN KEY (dept_id)
    REFERENCES departments(id)
    ON DELETE CASCADE
);

-- إضافة FK لجدول موجود
ALTER TABLE students
ADD FOREIGN KEY (dept_id)
  REFERENCES departments(id);
Advanced Subquery
الطلاب اللي درجتهم أعلى من المتوسط
SELECT name, grade
FROM   students
WHERE  grade > (
  SELECT AVG(grade) FROM students
);
Functions String / Date Functions 
UPPER(name)          -- كبير
LOWER(name)          -- صغير
LENGTH(name)         -- عدد الحروف
CONCAT(name,' X')    -- دمج نصوص
TRIM(name)           -- حذف spaces
SUBSTRING(name,1,3) -- أول 3 حروف
ROUND(grade, 2)      -- تقريب
NOW()                -- الوقت الحالي
YEAR(NOW())          -- السنة الحالية
Example ترتيب الكلاوز في الجملة الكاملة 
SELECT   d.dept_name, AVG(s.grade) AS avg_grade, COUNT(*) AS total
FROM     students s
JOIN     departments d ON s.dept_id = d.id
WHERE    s.grade > 50
GROUP BY d.dept_name
HAVING   AVG(s.grade) > 70
ORDER BY avg_grade DESC
LIMIT    5;

-- الترتيب دائماً: SELECT → FROM → JOIN → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT
